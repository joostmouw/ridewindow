```dart
class FakeWeatherLoading extends WeatherNotifier {
  @override
  Future<List<HourlyForecast>> build() async {
    // Blijf in loading state door nooit te completen — gebruik een Completer
    await Completer<void>().future; // hangt oneindig
    return [];
  }
}

class FakeSlotsWithData extends SlotsNotifier {
  @override
  SlotsState build() {
    // Lees geen andere providers — return direct
    ref.watch(weatherProvider); // nodig om te compileren
    ref.watch(profileProvider);
    ref.watch(availabilityProvider);
    return SlotsLoaded([_makeSlot()]);
  }
}

class FakeSlotsEmpty extends SlotsNotifier {
  final SlotsEmptyReason reason;
  FakeSlotsEmpty(this.reason);
  @override SlotsState build() {
    ref.watch(weatherProvider);
    ref.watch(profileProvider);
    ref.watch(availabilityProvider);
    return SlotsLoaded(const [], reason: reason);
  }
}
```

OPMERKING: Als FakeNotifiers die ref.watch aanroepen problemen geven omdat de upstream providers ook gefaked moeten worden — vereenvoudig: override slotsProvider direct met een statische SlotsState zonder ref.watch:

```dart
class FakeStaticSlots extends SlotsNotifier {
  final SlotsState _state;
  FakeStaticSlots(this._state);
  @override SlotsState build() => _state;
}
```
(SlotsNotifier is een synchrone Notifier — build() retourneert direct SlotsState. Maar build() in de echte implementatie roept ref.watch aan. Als we de upstream providers niet overriden, probeert Riverpod die te initialiseren. Override daarom ook weatherProvider, profileProvider, availabilityProvider met simpele fakes.)
