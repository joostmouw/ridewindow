`_handleNext`:
```dart
Future<void> _handleNext() async {
  if (_selected == null) return; // knop enabled maar noop als niets geselecteerd
  if (_selected == AvailabilityPreset.custom) {
    context.go('/availability');
    return;
  }
  // Bereken weekStart (aanstaande maandag of huidige als vandaag maandag is)
  final now = DateTime.now();
  final daysToMonday = (DateTime.monday - now.weekday + 7) % 7;
  final weekStart = DateTime(now.year, now.month, now.day)
      .subtract(Duration(days: now.weekday - DateTime.monday));
  // Gebruik huidige week-maandag (niet volgende week)
  final preset = buildPreset(_selected!, weekStart);
  await ref.read(availabilityProvider.notifier).seedPreset(preset);
  final prefs = await SharedPreferences.getInstance();
  await prefs.setBool('onboarding_complete', true);
  if (mounted) context.go('/home');
}
```
