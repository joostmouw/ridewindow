}

Riverpod 3.0 AsyncNotifier patroon (code-gen):
```dart
part 'my_notifier.g.dart';

@riverpod
class MyNotifier extends _$MyNotifier {
  @override
  Future<MyState> build() async {
    // laad initiële state
    return MyState();
  }

  Future<void> update(MyState next) async {
    // schrijf naar opslag
    state = AsyncData(next);
  }
}
```
