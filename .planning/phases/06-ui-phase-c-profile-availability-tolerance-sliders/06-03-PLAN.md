        BlockType.work   => const Color(0xFFB0BEC5),
        BlockType.custom => const Color(0xFFFF9800),
        null             => Colors.white,
      };

    // Tap: sla werk-blokken over (D-06-06)
    void _onCellTap(DateTime key, Map<DateTime, BlockType> blocked, WidgetRef ref) {
